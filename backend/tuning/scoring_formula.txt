SCORING FORMULA:

Core Skills Weight: 70%
Preferred Skills Weight: 20%
Project / Learning Bonus: 10%

core_score = matched_core / total_core
preferred_score = matched_preferred / total_preferred

raw_score = 
(core_score * 70) + 
(preferred_score * 20) + 
project_bonus

project_bonus:
+5 if ML project exists
+5 if NLP project exists

final_score = clamp(raw_score, 0, 100)
